Как это работает

Мы идем по массиву и копим сумму. Если сумма на предыдущем шаге была настолько плохой, что выгоднее начать заново с текущего числа — мы делаем "сброс".

Ячейка с белой рамкой — это тот самый "минимум", после которого мы начали новую, более успешную цепочку.

// JavaScript
function kadane(arr) {
  let curSum = arr[0], maxSum = arr[0];
  for(let i = 1; i < arr.length; i++) {
    curSum = Math.max(arr[i], curSum + arr[i]);
    maxSum = Math.max(maxSum, curSum);
  }
  return maxSum;
}
