Binary Heap (Max-Heap)

A binary heap is a complete binary tree where each node is greater than or equal to its children.

  • Sift Up: Restores property after insertion at the end.
  • Sift Down: Restores property after root extraction.
  • Array Mapping: Node at i has children at 2i+1 and 2i+2.
void siftUp(int i) {
    while (i > 0 && heap[(i-1)/2] < heap[i]) {
        swap(heap[i], heap[(i-1)/2]);
        i = (i-1)/2;
    }
}

void siftDown(int i) {
    int maxIdx = i;
    int l = 2*i + 1, r = 2*i + 2;
    if (l < heap.size() && heap[l] > heap[maxIdx]) maxIdx = l;
    if (r < heap.size() && heap[r] > heap[maxIdx]) maxIdx = r;
    if (i != maxIdx) {
        swap(heap[i], heap[maxIdx]);
        siftDown(maxIdx);
    }
}
