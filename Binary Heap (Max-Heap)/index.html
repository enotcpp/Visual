Binary Heap (Max-Heap)

Бинарная куча — это структура данных, представляющая собой полное бинарное дерево, где значение каждого родителя больше или равно значениям его детей.

  • Sift Up: Используется при вставке нового элемента в конец.
  • Sift Down: Используется при удалении корня (максимума).
  • Array Storage: Узел i имеет детей 2i+1 и 2i+2.
void siftUp(int i) {
    while (i > 0 && heap[(i-1)/2] < heap[i]) {
        swap(heap[i], heap[(i-1)/2]);
        i = (i-1)/2;
    }
}

void siftDown(int i) {
    int maxIdx = i;
    int l = 2*i + 1, r = 2*i + 2;
    if (l < heap.size() && heap[l] > heap[maxIdx]) maxIdx = l;
    if (r < heap.size() && heap[r] > heap[maxIdx]) maxIdx = r;
    if (i != maxIdx) {
        swap(heap[i], heap[maxIdx]);
        siftDown(maxIdx);
    }
}
